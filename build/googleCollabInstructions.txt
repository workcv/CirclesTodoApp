    'source.include_exts = py,png,jpg,kv,atlas':

        'android.permissions = READ_EXTERNAL_STORAGE,WRITE_EXTERNAL_STORAGE,MANAGE_EXTERNAL_STORAGE',
    'requirements = python3,kivy':
        'requirements = python3,kivy,android',
    '#android.api = 31':            'android.api = 33',
    '#android.minapi = 21':         'android.minapi = 21',
    '#android.archs = arm64-v8a, armeabi-v7a':
        'android.archs = arm64-v8a',
}

for old, new in replacements.items():
    if old in spec:
        spec = spec.replace(old, new)
        print(f'OK: {new[:50]}')
    else:
        print(f'NOT FOUND: {old[:50]}')

open('buildozer.spec', 'w').write(spec)
print('\nDone. Current relevant lines:')
for line in open('buildozer.spec'):
    if any(k in line for k in ['title','package','permissions','requirements','android.api','archs']):
        print(' ', line.strip())
